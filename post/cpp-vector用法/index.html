#include <iostream>
#include <vector>

int main() {
	// 定义
	std::vector<int> nums = {1,2,3};
	std::vector<std::string> names;
	// 增加
	nums.push_back(4);
	nums.insert(nums.begin()+1,99);

	names.push_back("Alice");
	// 查询
	for(auto it = nums.begin();it != nums.end();++it)
		std::cout << *it << std::endl;
	std::cout << std::endl;
	// 删除
	nums.erase(nums.begin());
	for(auto it = nums.begin();it != nums.end();++it)
		std::cout << *it << std::endl;
	std::cout << "vector大小:" << nums.size() << std::endl;
	std::cout << "是否为空:" << nums.empty() << std::endl;
	nums.clear();
	std::cout << "是否为空:" << nums.empty() << std::endl;
	return 0;
}

#include <iostream>
using namespace std;
template<typename T>
class Vector {
private:
	T* data;
	size_t size;
public:
	class iterator {
	private:
		T* ptr;
	public:
		explicit iterator(T* p = nullptr) : ptr(p) {}

		T& operator*() { return *ptr; }
		iterator& operator++() {++ptr; return *this; }
		bool operator!=(const iterator& other) const {
			return ptr != other.ptr;
		}
	};
	iterator begin() { return iterator(data); }
	iterator end() { return iterator(data+size);}

	Vector(size_t n = 0) : data(new T[n]),size(n) {}
	~Vector() { delete[] data; }
};
int main() {
	Vector<int> vec(3);
	for (auto it=vec.begin();it != vec.end(); ++it) {
		*it = 42;
	}
	cout << *vec.begin() << endl;
}
