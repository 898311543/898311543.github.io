    AVCodec *codec;
    av_log_set_level(AV_LOG_DEBUG);
    if(argc < 3){
        printf("Usage: %s <file> <codecname>\n", argv[0]);
        goto _ERROR;

    if(!codec){
        av_log(NULL,AV_LOG_ERROR,"coundn't find codec %s",codecname);
        goto _ERROR;
    }

    _ERROR:

        return 0;
}
